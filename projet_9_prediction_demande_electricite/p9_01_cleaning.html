# Turn strings into ints

def conversion_numerique(elt):
    elt = list(elt.split(','))
    elt = ".".join(elt)
    return elt

for i in liste_mois:
    x = map(conversion_numerique, dju[i])
    dju[i]= list(x)
    
dju[liste_mois] = dju[liste_mois].astype(float)
